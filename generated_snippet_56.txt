import streamlit as st
from ui.point_editor import render_point_classification_ui

def main():
    st.set_page_config(page_title="UR5e Software Suite", layout="wide")
    
    st.sidebar.title("Navigation")
    app_mode = st.sidebar.selectbox("Choose Module", ["Real-time Control", "Process Editor"])

    if app_mode == "Process Editor":
        render_point_classification_ui()
    else:
        st.info("Please select the Process Editor to manage robot programs.")

if __name__ == "__main__":
    main()
