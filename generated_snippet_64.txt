import plotly.graph_objects as go
import plotly.express as px
import pandas as pd
import numpy as np

class PerformanceVisualizer:
    @staticmethod
    def create_3d_trajectory_view(initial_traj, optimized_traj):
        """Generates a 3D overlay of initial vs optimized paths."""
        fig = go.Figure()

        # Initial Trajectory
        fig.add_trace(go.Scatter3d(
            x=[p[0] for p in initial_traj],
            y=[p[1] for p in initial_traj],
            z=[p[2] for p in initial_traj],
            mode='lines+markers',
            name='Initial Path',
            line=dict(color='#FF4B4B', width=4, dash='dash'),
            marker=dict(size=3, opacity=0.5)
        ))

        # Optimized Trajectory
        fig.add_trace(go.Scatter3d(
            x=[p[0] for p in optimized_traj],
            y=[p[1] for p in optimized_traj],
            z=[p[2] for p in optimized_traj],
            mode='lines+markers',
            name='Optimized Path',
            line=dict(color='#00CC96', width=6),
            marker=dict(size=4)
        ))

        fig.update_layout(
            scene=dict(xaxis_title='X (m)', yaxis_title='Y (m)', zaxis_title='Z (m)'),
            margin=dict(l=0, r=0, b=0, t=30),
            legend=dict(yanchor="top", y=0.99, xanchor="left", x=0.01)
        )
        return fig

    @staticmethod
    def create_saving_heatmap(segments_df):
        """Generates a heatmap showing time savings per segment."""
        # Reshape data for heatmap
        segments_df['Saving %'] = (segments_df['saving'] / segments_df['initial_time']) * 100
        
        fig = px.bar(
            segments_df,
            x='name',
            y='Saving %',
            color='Saving %',
            text=segments_df['saving'].apply(lambda x: f"{x}s"),
            color_continuous_scale='RdYlGn',
            title="Time Saving Efficiency per Segment"
        )
        fig.update_layout(xaxis_title="Process Segment", yaxis_title="Saving Percentage (%)")
        return fig
