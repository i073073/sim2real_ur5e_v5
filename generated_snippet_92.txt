import torch
import torch.nn as nn
from diffusers.schedulers.scheduling_ddpm import DDPMScheduler

class DiffusionPolicy(nn.Module):
    """
    UR5e의 정밀 제어를 위한 Diffusion Policy 신경망.
    현재 상태(Observation)를 조건(Condition)으로 하여 미래의 Action 시퀀스를 생성함.
    """
    def __init__(self, obs_dim=12, action_dim=6, hidden_dim=256):
        super().__init__()
        self.obs_encoder = nn.Sequential(
            nn.Linear(obs_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim)
        )
        
        # Denoising Network (간략화된 MLP 구조)
        self.model = nn.Sequential(
            nn.Linear(hidden_dim + action_dim, hidden_dim * 2),
            nn.ReLU(),
            nn.Linear(hidden_dim * 2, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, action_dim)
        )
        
        self.noise_scheduler = DDPMScheduler(num_train_timesteps=50)

    def forward(self, obs, noisy_action, timestep):
        # 1. Observation 인코딩
        obs_feat = self.obs_encoder(obs)
        
        # 2. Condition과 Noisy Action 결합
        combined = torch.cat([obs_feat, noisy_action], dim=-1)
        
        # 3. 노이즈 예측 (Predicting the noise added to the action)
        return self.model(combined)

    def sample_action(self, obs):
        """추론 단계: 노이즈로부터 최적의 Action 시퀀스 생성"""
        device = obs.device
        action = torch.randn((obs.shape[0], 6)).to(device)
        
        for t in self.noise_scheduler.timesteps:
            with torch.no_grad():
                noise_pred = self.forward(obs, action, t)
                action = self.noise_scheduler.step(noise_pred, t, action).prev_sample
        return action
