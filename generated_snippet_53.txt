import json
import csv
import io
from src.core.models import ProcessPoint, RobotProcess

class ProcessParser:
    @staticmethod
    def from_json(file_content: str) -> RobotProcess:
        """Parses a JSON formatted process file."""
        data = json.loads(file_content)
        return RobotProcess(**data)

    @staticmethod
    def from_csv(file_content: str) -> RobotProcess:
        """Parses a CSV formatted process file. 
        Expected headers: id, command_type, q0, q1, q2, q3, q4, q5, velocity, acceleration
        """
        f = io.StringIO(file_content)
        reader = csv.DictReader(f)
        points = []
        for row in reader:
            points.append(ProcessPoint(
                id=int(row['id']),
                command_type=row['command_type'],
                target=[float(row[f'q{i}']) for i in range(6)],
                velocity=float(row.get('velocity', 0.25)),
                acceleration=float(row.get('acceleration', 1.2)),
                classification="Mandatory"
            ))
        return RobotProcess(name="Imported CSV", version="1.0", points=points)

    @staticmethod
    def export_to_json(process: RobotProcess) -> str:
        """Exports the filtered/classified process back to JSON."""
        # Filter out points marked for deletion
        filtered_points = [p for p in process.points if p.classification != "Delete"]
        process.points = filtered_points
        return process.json(indent=4)
