import json
import time
import numpy as np
import genesis as gs
from src.mlops.schemas import JointDataset
from src.analysis.motion_comparator import MotionComparator

class URReplayEngine:
    def __init__(self, sim_instance, config_path="config/replay_config.yaml"):
        self.sim = sim_instance # Genesis UR5eSimulator 인스턴스
        with open(config_path, 'r') as f:
            import yaml
            self.cfg = yaml.safe_load(f)
        
        self.comparator = MotionComparator()
        self.replay_data = []

    def load_session(self, file_path):
        """JSON 로그 데이터 로드"""
        with open(file_path, 'r') as f:
            data = json.load(f)
            dataset = JointDataset(**data)
            self.replay_data = dataset.data_points
        print(f"Loaded {len(self.replay_data)} points from {file_path}")

    def run_replay(self):
        """시뮬레이션 내 데이터 재생 및 오차 기록"""
        if not self.replay_data:
            return

        print("Starting Replay and Comparison...")
        results = []

        for i, point in enumerate(self.replay_data):
            # 1. 시뮬레이터에 목표 관절값 주입 (Command)
            target_q = np.array(point.joints)
            self.sim.robot.set_dofs_position(target_q)
            
            # 2. 물리 엔진 스텝 진행
            self.sim.scene.step()
            
            # 3. 시뮬레이션 결과값 획득 (Simulated)
            sim_q = self.sim.robot.get_dofs_position()
            sim_tcp = self.sim.get_tcp_pose() # 시뮬레이터 내 TCP 계산 함수
            
            # 4. 오차 계산 및 기록
            error_entry = self.comparator.calculate_step_error(
                target_q, sim_q, point.timestamp
            )
            results.append(error_entry)

            # 5. 시각화 업데이트
            if i % 5 == 0:
                self.sim.scene.viewer.render()

        # 최종 분석 리포트 생성
        self.comparator.generate_summary(results)
