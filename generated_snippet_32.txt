import mlflow
import os

class MLOpsTracker:
    def __init__(self, experiment_name="UR5e_Data_Collection"):
        mlflow.set_experiment(experiment_name)

    def log_dataset_collection(self, file_path, success_count, fail_count):
        """데이터 수집 세션을 MLflow에 기록"""
        with mlflow.start_run(run_name=f"Data_Batch_{os.path.basename(file_path)}"):
            # 1. 파라미터 기록
            mlflow.log_param("storage_path", file_path)
            mlflow.log_param("raw_file_name", os.path.basename(file_path))
            
            # 2. 지표 기록 (품질 지표)
            total = success_count + fail_count
            success_rate = (success_count / total) * 100 if total > 0 else 0
            mlflow.log_metric("data_points_count", success_count)
            mlflow.log_metric("invalid_points_count", fail_count)
            mlflow.log_metric("data_quality_score", success_rate)

            # 3. 데이터셋 아티팩트 등록
            mlflow.log_artifact(file_path, artifact_path="datasets")
            
            print(f"MLflow Tracking Complete. Run ID: {mlflow.active_run().info.run_id}")
