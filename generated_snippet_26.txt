import time
import numpy as np
from src.hardware.ur_command_library import UR5eCommandLibrary
from src.utils.data_logger import JointDataLogger

def test_rtde_connection_and_latency(robot_ip="127.0.0.1"):
    """RTDE 연결 안정성 및 통신 지연 시간 측정"""
    print(f"\n[Test] Connecting to URSim at {robot_ip}...")
    client = UR5eCommandLibrary(robot_ip)
    
    if not client.is_connected:
        print("Result: FAILED - Could not connect to robot.")
        return False

    latencies = []
    for _ in range(100):
        start = time.perf_counter()
        _ = client.get_actual_q()
        latencies.append(time.perf_counter() - start)
    
    avg_latency = np.mean(latencies) * 1000 # ms
    max_latency = np.max(latencies) * 1000 # ms
    
    print(f"Result: PASSED")
    print(f"Average Latency: {avg_latency:.2f} ms")
    print(f"Max Latency: {max_latency:.2f} ms")
    
    client.disconnect()
    return avg_latency < 2.0 # RTDE 500Hz 기준 2ms 이내 권장

def test_logging_frequency_consistency():
    """10Hz 데이터 로깅 주기가 정확한지 검증"""
    from src.hardware.ur_command_library import UR5eCommandLibrary
    client = UR5eCommandLibrary("127.0.0.1")
    logger = JointDataLogger(client, frequency_hz=10, log_dir="tests/logs")
    
    print("\n[Test] Verifying 10Hz Logging Consistency...")
    logger.start()
    time.sleep(2.0) # 2초간 데이터 수집 (약 20개 샘플 기대)
    logger.stop()
    
    # 로그 파일 확인
    import glob
    import pandas as pd
    list_of_files = glob.glob('tests/logs/*.csv')
    latest_file = max(list_of_files, key=lambda x: x)
    df = pd.read_csv(latest_file)
    
    # 시간 간격 계산
    intervals = df['timestamp'].diff().dropna()
    avg_interval = intervals.mean()
    jitter = intervals.std()
    
    print(f"Sample Count: {len(df)}")
    print(f"Average Interval: {avg_interval:.4f}s (Target: 0.1000s)")
    print(f"Jitter (Std Dev): {jitter:.6f}s")
    
    assert 0.09 <= avg_interval <= 0.11
    client.disconnect()

if __name__ == "__main__":
    test_rtde_connection_and_latency()
    test_logging_frequency_consistency()
