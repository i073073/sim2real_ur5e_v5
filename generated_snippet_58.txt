import numpy as np

class ProcessObjective:
    def __init__(self, config):
        self.cfg = config

    def calculate_segment_time(self, p1, p2, v_max, a_max):
        """
        두 점 사이의 이동 시간을 사다리꼴 속도 프로파일 기반으로 추정
        t = (dist / v) + (v / a)
        """
        dist = np.linalg.norm(np.array(p2) - np.array(p1))
        if dist == 0: return 0
        
        # 가속 구간과 정속 구간을 고려한 최소 시간 계산
        t_acc = v_max / a_max
        dist_acc = 0.5 * a_max * (t_acc**2)
        
        if dist > 2 * dist_acc:
            t_const = (dist - 2 * dist_acc) / v_max
            return 2 * t_acc + t_const
        else:
            return 2 * np.sqrt(dist / a_max)

    def evaluate_constraints(self, trajectory):
        """제약 조건 위반 여부 확인 및 페널티 계산"""
        penalty = 0
        # 1. 관절 한계 및 속도 한계 체크
        # 2. 충돌 가능성 체크 (Distance to Obstacles)
        # 3. Jerk(가속도 변화율) 체크
        return penalty

    def fitness_function(self, sequence, points_data):
        """GA용 적합도 함수: 1 / (Total Time + Penalty)"""
        total_time = 0
        for i in range(len(sequence) - 1):
            p1 = points_data[sequence[i]].target
            p2 = points_data[sequence[i+1]].target
            v = points_data[sequence[i+1]].velocity
            a = points_data[sequence[i+1]].acceleration
            
            total_time += self.calculate_segment_time(p1, p2, v, a)
        
        penalty = self.evaluate_constraints(sequence)
        return 1.0 / (total_time + penalty + 1e-6)
