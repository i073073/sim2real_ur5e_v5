import numpy as np
import pandas as pd
from scipy.spatial.distance import euclidean
from fastdtw import fastdtw

class S2RValidator:
    """URSim 데이터와 s2r 데이터를 비교 분석하는 QA 툴"""
    
    def __init__(self, config):
        self.cfg = config

    def calculate_metrics(self, ref_data, sim_data):
        """
        ref_data: URSim 데이터 (Pandas DataFrame)
        sim_data: 시뮬레이션 데이터 (Pandas DataFrame)
        """
        results = {}
        
        # 1. Joint RMSE 계산
        q_ref = ref_data[[f'q{i}' for i in range(6)]].values
        q_sim = sim_data[[f'q{i}' for i in range(6)]].values
        
        joint_rmse = np.sqrt(np.mean((q_ref - q_sim)**2, axis=0))
        results['joint_rmse_avg'] = np.mean(joint_rmse)
        results['joint_rmse_per_joint'] = joint_rmse.tolist()

        # 2. TCP 위치 오차 (Euclidean Distance)
        p_ref = ref_data[['x', 'y', 'z']].values
        p_sim = sim_data[['x', 'y', 'z']].values
        
        pos_errors = [euclidean(p_ref[i], p_sim[i]) for i in range(len(p_ref))]
        results['pos_error_max_mm'] = np.max(pos_errors) * 1000
        results['pos_error_avg_mm'] = np.mean(pos_errors) * 1000

        # 3. DTW (Dynamic Time Warping) 경로 유사도
        distance, _ = fastdtw(q_ref, q_sim, dist=euclidean)
        results['dtw_distance'] = distance

        return results

    def evaluate_pass_fail(self, metrics):
        """설정된 임계값에 따른 합격 여부 판정"""
        report = []
        is_pass = True

        if metrics['pos_error_avg_mm'] > self.cfg['cartesian_space']['pos_error_limit_mm']:
            report.append(f"FAIL: Avg Pos Error {metrics['pos_error_avg_mm']:.2f}mm exceeds limit.")
            is_pass = False
        
        if metrics['joint_rmse_avg'] > self.cfg['joint_space']['rmse_limit']:
            report.append(f"FAIL: Joint RMSE {metrics['joint_rmse_avg']:.4f} exceeds limit.")
            is_pass = False

        if is_pass:
            report.append("SUCCESS: All metrics are within thresholds.")
            
        return is_pass, report
