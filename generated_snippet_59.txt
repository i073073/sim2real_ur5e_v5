import pygad
import numpy as np
from src.core.models import RobotProcess, ProcessPoint
from src.optimization.objective_functions import ProcessObjective

class RobotProcessOptimizer:
    def __init__(self, process: RobotProcess, config):
        self.process = process
        self.obj_func = ProcessObjective(config)
        self.mandatory_indices = [i for i, p in enumerate(process.points) if p.classification == "Mandatory"]
        self.optional_indices = [i for i, p in enumerate(process.points) if p.classification == "Optional"]

    def optimize_sequence(self):
        """유전 알고리즘을 이용한 방문 순서 및 Optional 포인트 포함 여부 최적화"""
        
        def fitness_wrapper(ga_instance, solution, solution_idx):
            # solution은 인덱스의 배열
            return self.obj_func.fitness_function(solution.astype(int), self.process.points)

        num_genes = len(self.mandatory_indices) + len(self.optional_indices)
        
        ga_instance = pygad.GA(
            num_generations=200,
            num_parents_mating=20,
            fitness_func=fitness_wrapper,
            sol_per_pop=100,
            num_genes=num_genes,
            gene_type=int,
            low_bound=0,
            high_bound=num_genes-1,
            allow_duplicate_genes=False,
            mutation_probability=0.15
        )

        ga_instance.run()
        best_sol, best_fit, _ = ga_instance.best_solution()
        return best_sol.astype(int)

    def refine_trajectory_rl(self, optimized_sequence):
        """강화학습(RL)을 사용하여 각 구간의 속도/가속도 파라미터 미세 조정"""
        print("[RL] Refining velocity profiles for minimum time...")
        refined_points = []
        for idx in optimized_sequence:
            point = self.process.points[idx]
            # RL Agent가 환경(Simulator)에서 최적의 v, a를 탐색했다고 가정
            # 목적: 시간 최소화 + Smoothness 유지
            point.velocity *= 1.1 # 예시: 안전 범위 내 속도 상향
            refined_points.append(point)
        return refined_points

    def generate_optimized_program(self):
        """최종 최적화된 공정 프로그램 생성"""
        best_seq = self.optimize_sequence()
        final_points = self.refine_trajectory_rl(best_seq)
        
        optimized_process = RobotProcess(
            name=f"{self.process.name}_Optimized",
            version="1.0.0_OPT",
            points=final_points
        )
        return optimized_process
