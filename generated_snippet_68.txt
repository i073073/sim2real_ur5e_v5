    def test_ga_optimization_efficiency(self):
        """유전 알고리즘의 경로 최적화 성능 검증"""
        waypoints = np.array([
            [0, 0, 0], [1, 1, 1], [0.5, 0.5, 0.5], [0.2, 0.8, 0.3]
        ])
        optimizer = TrajectoryOptimizerGA(waypoints)
        
        # 초기 랜덤 경로 거리 계산
        initial_dist = sum(np.linalg.norm(waypoints[i+1] - waypoints[i]) for i in range(len(waypoints)-1))
        
        # 최적화 실행
        best_seq = optimizer.optimize()
        optimized_waypoints = waypoints[best_seq]
        optimized_dist = sum(np.linalg.norm(optimized_waypoints[i+1] - optimized_waypoints[i]) for i in range(len(optimized_waypoints)-1))
        
        # 검증: 최적화된 거리가 초기 거리보다 짧거나 같아야 함
        assert optimized_dist <= initial_dist
        improvement = ((initial_dist - optimized_dist) / initial_dist) * 100
        print(f"[QA] GA Optimization: PASSED (Improvement: {improvement:.2f}%)")

    def test_objective_constraint_penalty(self):
        """목적 함수의 제약 조건 페널티 작동 여부 검증"""
        obj = ProcessObjective(config={'optimization': {'weights': {'constraint_penalty': 100.0}}})
        
        # 정상 궤적 (페널티 0 가정)
        safe_traj = [0, 1, 2]
        # 비정상 궤적 (충돌 발생 가정)
        unsafe_traj = [0, 2, 1] 
        
        # Mock evaluate_constraints
        def mock_eval(traj): return 100.0 if traj == unsafe_traj else 0.0
        obj.evaluate_constraints = mock_eval
        
        fit_safe = obj.fitness_function(safe_traj, points_data=[]) # points_data 생략
        fit_unsafe = obj.fitness_function(unsafe_traj, points_data=[])
        
        # 검증: 위험한 궤적의 적합도(Fitness)가 현저히 낮아야 함
        assert fit_unsafe < fit_safe
        print("[QA] Constraint Penalty Logic: PASSED")
