from pydantic import BaseModel, Field, validator
from typing import List, Optional
from datetime import datetime
import numpy as np

class JointDataPoint(BaseModel):
    """UR5e 단일 시점 관절 데이터 모델"""
    timestamp: float = Field(..., description="Unix timestamp")
    joints: List[float] = Field(..., min_items=6, max_items=6, description="6-DOF joint positions in radians")
    velocities: List[float] = Field(..., min_items=6, max_items=6, description="6-DOF joint velocities")
    is_valid: bool = True
    error_msg: Optional[str] = None

    @validator('joints')
    def check_joint_limits(cls, v):
        # UR5e 일반적인 안전 한계 (radians)
        limits = [2*np.pi] * 6 
        for i, angle in enumerate(v):
            if abs(angle) > limits[i]:
                raise ValueError(f"Joint {i} exceeds physical limit: {angle}")
        return v

class JointDataset(BaseModel):
    """저장용 데이터셋 구조"""
    session_id: str
    robot_id: str = "UR5e_Sim_01"
    collected_at: str = datetime.now().isoformat()
    data_points: List[JointDataPoint]
    metadata: dict = {
        "frequency_hz": 10,
        "environment": "URSim"
    }
