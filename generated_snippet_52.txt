from pydantic import BaseModel, Field
from typing import List, Literal

class ProcessPoint(BaseModel):
    """Model representing a single waypoint in a robot process."""
    id: int
    command_type: Literal["moveJ", "moveL"]
    target: List[float] = Field(..., min_items=6, max_items=6)
    velocity: float = 0.25
    acceleration: float = 1.2
    classification: Literal["Mandatory", "Optional", "Delete"] = "Mandatory"
    description: str = ""

class RobotProcess(BaseModel):
    """Model representing a full sequence of robot commands."""
    name: str
    version: str
    points: List[ProcessPoint]
