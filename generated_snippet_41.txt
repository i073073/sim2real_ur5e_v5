import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset
import json
import numpy as np

def load_and_preprocess_data(file_path):
    with open(file_path, 'r') as f:
        data = json.load(f)
    
    points = data['data_points']
    inputs = []
    targets = []

    for i in range(len(points) - 1):
        # State: Current Joints (6) + Target (Dummy for now, 6)
        state = points[i]['joints'] + [0.4, 0.2, 0.3, 0, 3.14, 0] 
        # Action: Next Joints (6)
        action = points[i+1]['joints']
        
        inputs.append(state)
        targets.append(action)

    return torch.tensor(inputs, dtype=torch.float32), torch.tensor(targets, dtype=torch.float32)

def train():
    X, y = load_and_preprocess_data("data/raw/session_latest.json")
    dataset = TensorDataset(X, y)
    loader = DataLoader(dataset, batch_size=64, shuffle=True)

    model = UR5eMimicNet().to("cuda")
    optimizer = optim.Adam(model.parameters(), lr=0.0005)
    criterion = nn.MSELoss()

    print("Starting Mimic Model Training...")
    for epoch in range(100):
        model.train()
        total_loss = 0
        for batch_x, batch_y in loader:
            batch_x, batch_y = batch_x.to("cuda"), batch_y.to("cuda")
            optimizer.zero_grad()
            output = model(batch_x)
            loss = criterion(output, batch_y)
            loss.backward()
            optimizer.step()
            total_loss += loss.item()
        
        if epoch % 10 == 0:
            print(f"Epoch {epoch}, Loss: {total_loss/len(loader):.6f}")

    torch.save(model.state_dict(), "models/ur5e_mimic_v1.pth")
    print("Model Saved.")

if __name__ == "__main__":
    train()
